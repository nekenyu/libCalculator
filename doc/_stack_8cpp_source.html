28  StackItem::Ptr dereferenceVariable(const VariableSet& variables, StackItem::Ptr in) {
29  Variable::Ptr asVariable = std::dynamic_pointer_cast<Variable, StackItem>(in);
30  if(!asVariable) {
31  return in;
32  }
33 
34  std::set<std::string> varNames;
35 
36  StackItem::Ptr item = in;
37  while(item) {
38  Variable::Ptr itemAsVar = std::dynamic_pointer_cast<Variable, StackItem>(item);
39  if(!itemAsVar) {
40  return item;
41  }
42 
43  // Add the current name to the list we've been through, unless already
44  // present. If already present, we have a circle and cannot dereference
45  auto result = varNames.insert(itemAsVar->getName());
46  if(!result.second) {
47  return StackItem::Ptr();
48  }
49 
50  // Dereference curr. An unset reference will be caught by the while loop
51  item = variables.get(itemAsVar->getName());
52  }
53  return StackItem::Ptr();
54  }
