33 void Executive::process(StackOperatorFactory& factory, std::istream& input, std::ostream& output) {
34  output << toString() << endl;
35  while(true) {
36  std::string command;
37  input >> command;
38  if(!input) {
39  break;
40  }
41 
42  const std::string result = process(factory, command);
43  if(result.empty()) {
44  output << "User requested quit..." << endl;
45  return;
46  }
47 
48  output << ">> " << operationCount << ": " << result << endl;
49  output << toString() << endl;
50  }
51 }

57 std::string Executive::process(StackOperatorFactory& factory, const std::string& input) {
58  if(0 == input.compare("quit")) {
59  return "";
60  }
61 
62  if(0 == input.compare("help")) {
63  std::ostringstream os;
64  os << "Help" << endl;
65  doHelp(factory, os);
66  return os.str();
67  }
68 
69  StackOperator::Ptr op = factory.create(input);
70  if(!op) {
71  return std::string("Ignoring invalid input: \"") + input + "\"";
72  }
73 
74  return process(op);
75 }
